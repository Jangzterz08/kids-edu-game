**Step 3: Rewrite `client/src/main.jsx`**

Mount order: ErrorBoundary wraps App, Toaster is sibling to App inside ErrorBoundary (so toasts still show during normal operation but ErrorBoundary catches crashes):

```jsx
import { createRoot } from 'react-dom/client';
import { ErrorBoundary } from 'react-error-boundary';
import { Toaster } from 'sonner';
import ErrorFallback from './components/ui/ErrorFallback';
import App from './App.jsx';
import './index.css';

createRoot(document.getElementById('root')).render(
  <ErrorBoundary
    FallbackComponent={ErrorFallback}
    onError={(error, info) => {
      console.error('[ErrorBoundary]', error, info);
    }}
    onReset={() => window.location.assign('/')}
  >
    <App />
    <Toaster position="top-center" richColors />
  </ErrorBoundary>
);
```
